CREATE TABLE BUYER (SNO INT AUTO_INCREMENT NOT NULL, BUYER_NAME VARCHAR(150) NOT NULL, BUYER_ID VARCHAR(10) NOT NULL, EMAIL VARCHAR(100) NOT NULL, PHNO BIGINT NOT NULL, PASSWORD VARCHAR(100) NOT NULL, PRIMARY KEY(SNO,BUYER_ID));

CREATE TABLE SELLER (SNO INT AUTO_INCREMENT NOT NULL, SELLER_NAME VARCHAR(150) NOT NULL, SELLER_ID VARCHAR(10) NOT NULL, EMAIL VARCHAR(100) NOT NULL, PHNO BIGINT NOT NULL, PASSWORD VARCHAR(100) NOT NULL, PRIMARY KEY(SNO,SELLER_ID));
